content.phase
verifyText css=h2.CollectionListTopic-title.js-CollectionListTopic-title More on this topic
storeXpathCount xpath=//div[@class='Page-category-content']/ul[@class='CollectionListBoxes-list']/li MORE_ON_THIS
getEval LOG.warn("${MORE_ON_THIS} - MORE ON THIS");
verifyEval ${MORE_ON_THIS} >=  6 ? true : false true
store 1 i
store 6 view_more_click
store 2 view_more_click_ad_count
while ${MORE_ON_THIS} >= ${i}
getEval LOG.warn("== More on this cell[${i}] ==");
store xpath=//div[@class='Page-category-content']/ul[@class='CollectionListBoxes-list']/li[${i}]/div/a/img eval_item
storeAttribute xpath=//div[@class='Page-category-content']/ul[@class='CollectionListBoxes-list']/li[${i}]/div/a/img@src eval_data
verifyElementPresent ${eval_item}
getEval LOG.warn("   - More On: image: ${eval_data}");
store xpath=//div[@class='Page-category-content']/ul[@class='CollectionListBoxes-list']/li[${i}]/div/a/div/div/div[contains(@class, 'CollectionListBoxes-box-info-title')] eval_item
verifyText ${eval_item} regexpi:([a-z0-9\.\?\(\)\!\s\-\&]){2,40}
storeText ${eval_item} eval_data
getEval LOG.warn("   - More On: Title: ${eval_data}");
store xpath=//div[@class='Page-category-content']/ul[@class='CollectionListBoxes-list']/li[${i}]/div/a/div/div/div[contains(@class, 'CollectionListBoxes-box-author-label')]/span eval_item
verifyText ${eval_item} regexpi:([a-z0-9\.\s]){3,40}
storeText ${eval_item} eval_data
getEval LOG.warn("   - More On: Posted By: ${eval_data}");
gotoIf ${view_more_click} != ${i} click_view_more
gotoIf storedVars['MORE_ON_THIS'] <= 6 click_view_more
click //div[@class='CollectionListBoxes-button']/button[contains(@class, 'view-more-all')]
getEval LOG.warn("== More On: CLICKED View More Botton ==");
getEval storedVars['view_more_click']=6+storedVars['view_more_click']
verifyElementPresent xpath=//div[${view_more_click_ad_count}]/div[contains(@class, 'Ad-IframeWrap')]/iframe
getEval LOG.warn("   - More On: check AD after - click View More Botton");
label click_view_more
getEval storedVars['i']++;
endWhile
