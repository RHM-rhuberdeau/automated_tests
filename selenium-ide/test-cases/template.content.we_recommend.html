template.we_recommend
gotoIf storedVars['CURRENT_URL'].match(/.*\/p\/[0-9]{1,3}.*/g) && ! storedVars['CURRENT_URL'].match(/.*\/p\/1/g); after_we_recomment
verifyText css=h4.CollectionListBoxes-titleSub We Recommend
storeXpathCount xpath=//div[contains(@class, 'CollectionListWeRecommend')]/ul[@class='CollectionListBoxes-list']/li WE_RECOMMEND_COUNT
getEval LOG.warn("WE RECOMMEND ${WE_RECOMMEND_COUNT} Total");
verifyEval ${WE_RECOMMEND_COUNT} >=  1 ? true : false true
store 1 i
while ${WE_RECOMMEND_COUNT} >= ${i};
getEval LOG.warn("== We Recommend cell[${i}] ==");
store xpath=//ul[@class='CollectionListBoxes-list']/li[${i}]/div/a/img eval_item
storeAttribute xpath=//ul[@class='CollectionListBoxes-list']/li[${i}]/div/a/img@src eval_data
verifyElementPresent ${eval_item}
getEval LOG.warn("   - We Recommend: image: ${eval_data}");
store xpath=//ul[@class='CollectionListBoxes-list']/li[${i}]/div/a/div[@class='CollectionListBoxes-box-info']/div eval_item
verifyText ${eval_item} regexpi:([a-z0-9\.\?\(\)\!\s\-]){12,40}
storeText ${eval_item} eval_data
getEval LOG.warn("   - We Recommend: Title: " + storedVars['eval_data']);
getEval storedVars['i']++;
endWhile
label after_we_recomment
